cmake_minimum_required(VERSION 3.10)

# Define the project
project(OpenGL_GLFW_GLAD_Project)

# Set C++ Standard
set(CMAKE_CXX_STANDARD 17)

# Include directories for GLAD and GLFW (if needed)
include_directories(include)

# Add GLAD source file manually
set(GLAD_SOURCES src/glad.c)

# Add your other source files from src/ folder
file(GLOB PROJECT_SOURCES "src/*.cpp")

# Find OpenGL
find_package(OpenGL REQUIRED)

# Specify the path to GLFW's include directory and library
set(GLFW_INCLUDE_DIR ${CMAKE_SOURCE_DIR}/include/glfw)
set(GLFW_LIBRARY ${CMAKE_SOURCE_DIR}/lib/glfw3.lib)


# Add include directory for GLFW
include_directories(${GLFW_INCLUDE_DIR})

# Create the executable
add_executable(OpenGL_GLFW_GLAD_Project ${PROJECT_SOURCES} ${GLAD_SOURCES})

# Link the libraries (OpenGL, GLFW)
target_link_libraries(OpenGL_GLFW_GLAD_Project OpenGL::GL ${GLFW_LIBRARY})
